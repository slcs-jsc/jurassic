56  static char line[LEN];
57 
58  static double rtime[NLMAX], rz[NLMAX], rlon[NLMAX], rlat[NLMAX], obs_meas,
59  obs_sim, scl = 1.0, scl_old, scl_err, c0, c1, cov00, cov01, cov11,
60  sumsq, x[NMAX], x2[NMAX], y[NMAX], y_err[NMAX], y2[NMAX], y2_err[NMAX],
61  y2_sim[NMAX], y2_sim_err[NMAX], w2[NMAX], dt, tol, obs_err;
62 
63  static float rp[NLMAX], rt[NLMAX], rso2[NLMAX], rh2o[NLMAX],
64  ro3[NLMAX], robs[NLMAX];
65 
66  static int data, fit, i, ig, il, ip, it, itmax, n, nl, ndata[NMAX], nprof;
67 
68  static size_t mk, nk;
69 
70  /* Check arguments... */
71  if (argc < 5)
72  ERRMSG("Give parameters: <ctl> <prof> <inv> <kernel>");
73 
74  /* Read control parameters... */
75  read_ctl(argc, argv, &ctl);
76  dt = scan_ctl(argc, argv, "INVERT_DT", -1, "86400", NULL);
77  obs_err = scan_ctl(argc, argv, "INVERT_OBS_ERR", -1, "1.0", NULL);
78  data = (int) scan_ctl(argc, argv, "INVERT_DATA", -1, "2", NULL);
79  fit = (int) scan_ctl(argc, argv, "INVERT_FIT", -1, "3", NULL);
80  itmax = (int) scan_ctl(argc, argv, "INVERT_ITMAX", -1, "10", NULL);
81  tol = scan_ctl(argc, argv, "INVERT_TOL", -1, "1e-4", NULL);
82 
83  /* Check control parameters... */
84  if (ctl.ng != 4)
85  ERRMSG("Set NG = 4!");
86  if (strcmp(ctl.emitter[0], "SO2") != 0)
87  ERRMSG("Set EMITTER[0] = SO2!");
88  if (strcmp(ctl.emitter[1], "H2O") != 0)
89  ERRMSG("Set EMITTER[1] = H2O!");
90  if (strcmp(ctl.emitter[2], "O3") != 0)
91  ERRMSG("Set EMITTER[2] = O3!");
92  if (strcmp(ctl.emitter[3], "CO2") != 0)
93  ERRMSG("Set EMITTER[3] = CO2!");
94  if (ctl.nd != 2)
95  ERRMSG("Set ND = 2!");
96 
97  /* Set control parameters... */
98  ctl.write_bbt = 1;
99  ctl.write_matrix = 1;
100 
101  /* Set observation data... */
102  obs.nr = 1;
103  obs.obsz[0] = 705;
104 
105  /* ------------------------------------------------------------
106  Read profiles...
107  ------------------------------------------------------------ */
108 
109  /* Read profile data... */
110  printf("Read profile data: %s\n", argv[2]);
111 
112  /* Open file... */
113  if (!(in = fopen(argv[2], "r")))
114  ERRMSG("Cannot open file!");
115 
116  /* Read file... */
117  while (fgets(line, LEN, in)) {
118 
119  /* Read data... */
120  if (sscanf(line, "%lg %lg %lg %lg %g %g %g %g %g %g", &rtime[nl],
121  &rz[nl], &rlon[nl], &rlat[nl], &rp[nl], &rt[nl], &rso2[nl],
122  &rh2o[nl], &ro3[nl], &robs[nl]) != 10)
123  continue;
124  if ((++nl) > NLMAX)
125  ERRMSG("Too many profile data points!");
126  }
127 
128  /* Close files... */
129  fclose(in);
130 
131  /* ------------------------------------------------------------
132  Fit scaling factor for total mass...
133  ------------------------------------------------------------ */
134 
135  /* Iterations... */
136  for (it = 0; it < itmax; it++) {
137 
138  /* Init... */
139  atm.np = n = 0;
140  for (i = 0; i < NMAX; i++) {
141  ndata[i] = 0;
142  x[i] = y[i] = GSL_NAN;
143  }
144 
145  /* Loop over lines... */
146  for (il = 0; il < nl; il++) {
147 
148  /* Check for new profile... */
149  if ((rtime[il] != atm.time[0]
150  || rlon[il] != atm.lon[0]
151  || rlat[il] != atm.lat[0])
152  && atm.np > 0) {
153 
154  /* Call forward model... */
155  formod(&ctl, &atm, &obs);
156  obs_sim = obs.rad[0][0] - obs.rad[1][0];
157 
158  /* Get time index... */
159  i = (int) ((atm.time[0] - rtime[0]) / dt);
160  if (i < 0 && i >= NMAX)
161  ERRMSG("Time index out of range!");
162 
163  /* Get maxima... */
164  if (data == 1) {
165  x[i] = (gsl_finite(x[i]) ? GSL_MAX(x[i], obs_sim) : obs_sim);
166  y[i] = (gsl_finite(y[i]) ? GSL_MAX(y[i], obs_meas) : obs_meas);
167  y_err[i] = obs_err;
168  if (gsl_finite(x[i]) && gsl_finite(y[i]))
169  ndata[i] = 1;
170  }
171 
172  /* Get means... */
173  else if (data == 2) {
174  if (ndata[i] == 0) {
175  x[i] = obs_sim;
176  y[i] = obs_meas;
177  y_err[i] = POW2(obs_meas);
178  } else {
179  x[i] += obs_sim;
180  y[i] += obs_meas;
181  y_err[i] += POW2(obs_meas);
182  }
183  ndata[i]++;
184  }
185 
186  /* Calculate mean atmospheric profile... */
187  nprof++;
188  atm2.np = atm.np;
189  for (ip = 0; ip < atm.np; ip++) {
190  atm2.time[ip] += atm.time[ip];
191  atm2.z[ip] += atm.z[ip];
192  atm2.lon[ip] += atm.lon[ip];
193  atm2.lat[ip] += atm.lat[ip];
194  atm2.p[ip] += atm.p[ip];
195  atm2.t[ip] += atm.t[ip];
196  for (ig = 1; ig < ctl.ng; ig++)
197  atm2.q[ig][ip] += atm.q[ig][ip];
198  }
199 
200  /* Reset counter... */
201  atm.np = 0;
202  }
203 
204  /* Save data... */
205  obs_meas = robs[il];
206  atm.time[atm.np] = rtime[il];
207  atm.z[atm.np] = rz[il];
208  atm.lon[atm.np] = rlon[il];
209  atm.lat[atm.np] = rlat[il];
210  atm.p[atm.np] = rp[il];
211  atm.t[atm.np] = rt[il];
212  atm.q[0][atm.np] = rso2[il] * scl;
213  atm.q[1][atm.np] = rh2o[il];
214  atm.q[2][atm.np] = ro3[il];
215  atm.q[3][atm.np] = 371.789948e-6 + 2.026214e-6
216  * (atm.time[atm.np] - 63158400.) / 31557600.;
217  if ((++atm.np) > NP)
218  ERRMSG("Too many data points!");
219  }
220 
221  /* Calculate means... */
222  if (data == 2)
223  for (i = 0; i < NMAX; i++)
224  if (ndata[i] > 0) {
225  x[i] /= ndata[i];
226  y[i] /= ndata[i];
227  y_err[i] = sqrt(GSL_MAX(y_err[i] / ndata[i] - POW2(y[i]), 0.0))
228  / sqrt(ndata[i]); /* standard error! */
229  }
230 
231  /* Filter data... */
232  n = 0;
233  for (i = 0; i < NMAX; i++)
234  if (ndata[i] > 0 && gsl_finite(x[i]) && gsl_finite(y[i])
235  && gsl_finite(y_err[i])) {
236  x2[n] = x[i];
237  y2[n] = y[i];
238  y2_err[n] = y_err[i];
239  w2[n] = 1. / POW2(y_err[i]);
240  n++;
241  }
242 
243  /* Fit radiance data... */
244  if (fit == 1)
245  gsl_fit_mul(x2, 1, y2, 1, (size_t) n, &c1, &cov11, &sumsq);
246  else if (fit == 2)
247  gsl_fit_wmul(x2, 1, w2, 1, y2, 1, (size_t) n, &c1, &cov11, &sumsq);
248  else if (fit == 3)
249  gsl_fit_linear(x2, 1, y2, 1, (size_t) n, &c0, &c1, &cov00, &cov01,
250  &cov11, &sumsq);
251  else if (fit == 4)
252  gsl_fit_wlinear(x2, 1, w2, 1, y2, 1, (size_t) n, &c0, &c1, &cov00,
253  &cov01, &cov11, &sumsq);
254  else
255  ERRMSG("Check INVERT_FIT!");
256 
257  /* Get new scaling factor... */
258  scl_old = scl;
259  scl_err = scl * sqrt(cov11);
260  scl *= c1;
261 
262 
263  /* scl_err = scl * sqrt(cov11 / POW2(c1)); */
264 
265 
266  /* Write info... */
267  printf(" it= %d | scl= %g +/- %g | RMSE= %g\n",
268  it, scl, scl_err, sqrt(sumsq / n));
269 
270  /* Convergence test... */
271  if (fabs(2.0 * (scl - scl_old) / (scl + scl_old)) < tol)
272  break;
273  }
274 
275  /* ------------------------------------------------------------
276  Write inversion data...
277  ------------------------------------------------------------ */
278 
279  /* Write info... */
280  printf("Write inversion data: %s\n", argv[3]);
281 
282  /* Create file... */
283  if (!(out = fopen(argv[3], "w")))
284  ERRMSG("Cannot create file!");
285 
286  /* Write header... */
287  fprintf(out,
288  "# $1 = time (seconds since 2000-01-01T00:00Z)\n"
289  "# $2 = simulated SO2 index [K]\n"
290  "# $3 = scaled simulated SO2 index [K]\n"
291  "# $4 = error of scaled simulated SO2 index [K]\n"
292  "# $5 = observed SO2 index [K]\n"
293  "# $6 = error of observed SO2 index [K]\n\n");
294 
295  /* Write data... */
296  for (i = 0; i < n; i++) {
297 
298  /* Calculate scaled SO2 index... */
299  if (fit == 1 || fit == 2)
300  gsl_fit_mul_est(x2[i], c1, cov11, &y2_sim[i], &y2_sim_err[i]);
301  else if (fit == 3 || fit == 4)
302  gsl_fit_linear_est(x2[i], c0, c1, cov00, cov01, cov11, &y2_sim[i],
303  &y2_sim_err[i]);
304 
305  /* Write output... */
306  fprintf(out, "%.2f %g %g %g %g %g\n", rtime[0] + (i + 0.5) * dt,
307  x2[i], y2_sim[i], y2_sim_err[i], y2[i], y2_err[i]);
308  }
309 
310  /* Report scaling factor for total mass... */
311  fprintf(out, "\n");
312  fprintf(out, "# scl= %g +/- %g\n", scl, scl_err);
313  fprintf(out, "# c1= %g +/- %g\n", c1, sqrt(cov11));
314  if (fit == 3 || fit == 4) {
315  fprintf(out, "# c0= %g +/- %g\n", c0, sqrt(cov00));
316  fprintf(out, "# corr= %g\n", cov01 / (sqrt(cov00) * sqrt(cov11)));
317  }
318  fprintf(out, "# RMSE= %g\n", sqrt(sumsq / n));
319  fprintf(out, "# n= %d\n", n);
320 
321  /* Close files... */
322  fclose(out);
323 
324  /* ------------------------------------------------------------
325  Calculate kernel...
326  ------------------------------------------------------------ */
327 
328  /* Set atmospheric data... */
329  for (ip = 0; ip < atm2.np; ip++) {
330  atm2.time[ip] /= nprof;
331  atm2.z[ip] /= nprof;
332  atm2.lon[ip] /= nprof;
333  atm2.lat[ip] /= nprof;
334  atm2.p[ip] /= nprof;
335  atm2.t[ip] /= nprof;
336  for (ig = 0; ig < ctl.ng; ig++)
337  atm2.q[ig][ip] /= nprof;
338  }
339 
340  /* Get sizes... */
341  nk = atm2x(&ctl, &atm2, NULL, NULL, NULL);
342  mk = obs2y(&ctl, &obs, NULL, NULL, NULL);
343 
344  /* Allocate... */
345  k = gsl_matrix_alloc(mk, nk);
346 
347  /* Compute kernel matrix... */
348  kernel(&ctl, &atm2, &obs, k);
349 
350  /* Write matrix to file... */
351  write_matrix(NULL, argv[4], &ctl, k, &atm2, &obs, "y", "x", "r");
352 
353  /* Free... */
354  gsl_matrix_free(k);
355 
356  return EXIT_SUCCESS;
