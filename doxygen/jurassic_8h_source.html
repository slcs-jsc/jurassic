85#define LIN(x0, y0, x1, y1, x) \
86 ((y0)+((y1)-(y0))/((x1)-(x0))*((x)-(x0)))
87
89#define LOGX(x0, y0, x1, y1, x) \
90 (((x)/(x0)>0 && (x1)/(x0)>0) \
91 ? ((y0)+((y1)-(y0))*log((x)/(x0))/log((x1)/(x0))) \
92 : LIN(x0, y0, x1, y1, x))
93
95#define LOGY(x0, y0, x1, y1, x) \
96 (((y1)/(y0)>0) \
97 ? ((y0)*exp(log((y1)/(y0))/((x1)-(x0))*((x)-(x0)))) \
98 : LIN(x0, y0, x1, y1, x))
